break line: ctrl + e, ctrl + j / ctrl + m

save file: ctrl + x, ctrl + s

exit: ctrl + x, k, enter

terminate input: ctrl + g

cut: ctrl + w

copy: alt + w

paste: ctrl + y

cut to end: ctrl + k

delete one character after: ctrl + d

delete one character before: delete

up: ctrl + up

down: ctrl + down

left: ctrl + b

right: ctrl + f

start line: ctrl + a

end line: ctrl + e

exit: ctrl + x, ctrl + c

open file: ctrl + x, ctrl + f

revoke: ctrl + x, u

maxium emacs: ctrl + z

down query: ctrl + s

query and replace: alt + %
